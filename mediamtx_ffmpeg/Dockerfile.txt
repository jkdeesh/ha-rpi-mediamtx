ARG BUILD_FROM=debian:bookworm-slim
FROM $BUILD_FROM

# Install deps
RUN apt-get update && apt-get install -y \
    ffmpeg \
    wget \
    ca-certificates \
    && rm -rf /var/lib/apt/lists/*

# Install MediaMTX (latest arm64)
RUN MEDIAMTX_VERSION=$(curl -s https://api.github.com/repos/bluenviron/mediamtx/releases/latest | grep '"tag_name"' | sed -E 's/.*"([^"]+)".*/\1/') \
    && wget https://github.com/bluenviron/mediamtx/releases/download/${MEDIAMTX_VERSION}/mediamtx_${MEDIAMTX_VERSION}_linux_arm64v8.tar.gz \
    && tar -xzf mediamtx_${MEDIAMTX_VERSION}_linux_arm64v8.tar.gz -C /usr/local/bin/ mediamtx \
    && chmod +x /usr/local/bin/mediamtx \
    && rm mediamtx_${MEDIAMTX_VERSION}_linux_arm64v8.tar.gz

# Copy scripts
COPY run.sh /
RUN chmod a+x /run.sh

CMD [ "/run.sh" ]
